    K8S配置管理-Configmap

    # 云原生要素-配置分离

    ConfigMap：存储明文配置

    Secret:存储密文，敏感配置，各种密码；

    配置更新直接同步容器，热加载，无需重启pod或者容器；镜像和配置分离，可单独修改发布；

    pod中可以通过valueFrom，envfrom来引用

    https://kubernetes.io/zh-cn/docs/concepts/configuration/configmap/

          env:
        # 定义环境变量
        - name: PLAYER_INITIAL_LIVES # 请注意这里和 ConfigMap 中的键名是不一样的
          valueFrom:
            configMapKeyRef:
              name: game-demo           # 这个值来自 ConfigMap
              key: player_initial_lives # 需要取值的键
        - name: UI_PROPERTIES_FILE_NAME
          valueFrom:
            configMapKeyRef:
              name: game-demo
              key: ui_properties_file_name

    推荐使用envFrom方式。

    # 使用envFrom批量生成环境变量

    [root@k8s-master01 configmap]# cat dp-envfrom-cm.yaml 
apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: dp-cm
  name: dp-cm
spec:
  replicas: 1
  selector:
    matchLabels:
      app: dp-cm
  template:
    metadata:
      labels:
        app: dp-cm
    spec:
      containers:
      - image: nginx
        name: nginx
        #cm gameenvcm里的变量全部输出
        envFrom:
        - configMapRef:
            name: gameenvcm 
          prefix: fromCm # 环境变量前缀
        env:
        - name: TEST_ENV
          value: testenv
        - name: LIVES

    # 自定义挂载权限及名称

    apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: dp-cm
  name: dp-cm
spec:
  replicas: 1
  selector:
    matchLabels:
      app: dp-cm
  strategy: {}
  template:
    metadata:
      labels:
        app: dp-cm
    spec:
      containers:
      - image: nginx
        name: nginx
        volumeMounts:
        - name: redis-conf-volumes
          mountPath: /etc/config
      volumes:
      - name: redis-conf-volumes
        configMap:
          name: cmfromfile
          items: 
          - key: redis.conf  # key名
            path: redis-conf.new  # 修改挂载到pod 容器中的文件名
            mode: 0644 #优先级高
          defaultMode: 0644  #  修改挂载权限
